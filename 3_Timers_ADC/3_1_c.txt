#include <avr/io.h>
#include <stdint.h>

uint8_t index=8;
uint8_t DC_VALUE=50;

const uint8_t duty_table[]={5,20,36,51,66,82,97,112,128,143,158,173,189,204,219,235,250};

void fortosi(uint8_t i){
	OCR1AL=duty_table[i];
	return;
}


int main(void){
	
	TCCR1A=	(1<<WGM10)|(1<<COM1A1); //non inverting fast PWM 8 bit
	TCCR1B=	(1<<WGM12)|(1<<CS10); //fast PWM, CLK
	DDRB=0b00000010; //exodos PB1
	PORTB |= (1<<PB4) | (1<<PB5); //pull-ups gia PB4, PB5
	OCR1AH=0;
	OCR1AL=128; //init exodos sto 50% DC
	
	while (1){
		
		if(!(PINB&(1<<PB5))){ //an PB5 patimeno
			while(!(PINB&(1<<PB5))); //debounce 
			if(index){
				index--;
				DC_VALUE-=6;
				fortosi(index);
			}
			
		}
		if(!(PINB&(1<<PB4))){ //an PB4 patimeno
			while(!(PINB&(1<<PB4))); //debounce
			if(index<16){
				index++;
				DC_VALUE+=6;
				fortosi(index);
			}
			
			
			
		}
		
		
		
	}
	
}
