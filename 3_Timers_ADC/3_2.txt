#include <avr/io.h>
#include <stdint.h>
#include <util/delay.h>

uint8_t index=8;
uint8_t DC_VALUE=50;

const uint8_t duty_table[]={5,20,36,51,66,82,97,112,128,143,158,173,189,204,219,235,250};
	
int ADCvalue=0; //int giati thelw 2 bytes
uint8_t i=16;

void fortosi(uint8_t i){
	OCR1AL=duty_table[i];
	return;
}


int main(void){
	
	TCCR1A=	(1<<WGM10)|(1<<COM1A1); //non inverting fast PWM 8 bit
	TCCR1B=	(1<<WGM12)|(1<<CS10); //fast PWM, CLK
	DDRB=0b00000010; //PB1 exodos, PB4, PB5 eisodoi
	DDRD=0b11111111; //PORTD exodoi
	PORTB=(1<<PB4)|(1<<PB5); //pull-ups
	
	OCR1AH=0;
	OCR1AL=128;
	
	//gia adc
	ADMUX=0b01000001; //select vref=5 to 01, ADLAR=0 gia right adjusted, 0001 gia pc1=pwb1
	ADCSRA=0b10000111; //aden=1 gia adc enable, adcs=0 gia no conversion, adie=0 gia disable adc interrupt, adps=111 fadc=125khz
	
	while (1){
		if(!(PINB&(1<<PB5))){
			while(!(PINB&(1<<PB5)));
			if(index){
				index--;
				DC_VALUE-=6;
				fortosi(index);
			}
			
		}
		if(!(PINB&(1<<PB4))){
			while(!(PINB&(1<<PB4)));
			if(index<16){
				index++;
				DC_VALUE+=6;
				fortosi(index);

			}
			
			
			
		}
		// adsc flag energopoisi gia xekinima metatropis
		ADCSRA|=(1<<ADSC);
		while(ADCSRA&(1<<ADSC));
		
		if((i--)>0){
			ADCvalue+=ADC;//prosthetw 16 fores
			_delay_ms(100);
		}
		if(i==0){
			ADCvalue=ADCvalue >> 4 ; //olisthisi  dexia 4 fores gia mesi timi metrisewn ousiastika diairei me to 16
			i=16;
			if(ADCvalue<=200){
				PORTD=(1<<PD0);
			}
			else if(ADCvalue>=200 && ADCvalue <=400){
				PORTD=(1<<PD1);
			}
			else if(ADCvalue>=400 && ADCvalue <=600){
				PORTD=(1<<PD2);
			}
			else if(ADCvalue>=600 && ADCvalue <=800){
				PORTD=(1<<PD3);
			}
			else{
				PORTD=(1<<PD4);
			}
			ADCvalue=0;
		}
	} //tou while einai ayto
	
} //tou main eina ayto
