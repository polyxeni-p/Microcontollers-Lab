.include "m328PBdef.inc"
.equ FOSC_MHZ=16
.equ DEL_mS=1000 ;delay 1000mS
.equ DEL_NU=FOSC_MHZ*DEL_mS

.org 0x0
rjmp reset

.org 0x2
rjmp ISR0

reset:

;Init Stack Pointer
	ldi r24, LOW(RAMEND)
	out SPL, r24
	ldi r24, HIGH(RAMEND)
	out SPH, r24

;Init PORTC as output
	ser r26
	out DDRC, r26

;Init PORTB as input
	clr r26
	out DDRB, r26

;Interrupt INT0 sti xamili stathmi
	ldi r24, (0<<ISC01)|(0<<ISC00) ;xamili stathmi, dld otan einai patimeno to koumpi
	sts EICRA, r24

;Enable the INT0 interrupt (PD2)
	ldi r24, (1<<INT0)
	out EIMSK, r24

	clr r24
	out PORTC, r24 ;svista led

	sei

;atermono loop
;counter from 1 to 31, output to PC5-PC1
loop1:
	clr r26
loop2:
	lsl r26			;gia display sta PC5-PC1
	out PORTC, r26	;exodos
	lsr r26			;epanafora

	ldi r24, low(DEL_NU)
	ldi r25, high(DEL_NU)
	rcall delay_mS	;kathisterisi 1s

	inc r26			;inc metriti

	cpi r26, 32		;check an eftasa 32
	breq loop1		;an nai epanafora metriti
	rjmp loop2

;routina gia kathisterisi 1000*F1+6 cycles
delay_mS:
	ldi r23, 249	;1c
loop_inn:
	dec r23			;1c
	nop				;1c
	brne loop_inn	;1c or 2c if branch

	sbiw r24, 1		;2c
	brne delay_mS	;1c or 2c if branch

	ret				;4c

;routina INT0
ISR0:
	push r24
	push r25
	push r26
	in r26, SREG
	push r26

	ldi r24, 0x0F ;00001111b
	in r25, PINB

	ldi r26, 4 ;counter
loop_4bits:
	lsr r25 ;rotate right
	sbrc r25, 0 ;skip tin epomeni entoli an to 1o bit einai 0 (patimeno) 
	lsr r24
	dec r26
	brne loop_4bits

	out PORTC, r24

	pop r26
	out SREG, r26
	pop r26
	pop r25
	pop r24

	reti
