#define PCA9555_0_ADDRESS 0x40 //A0=A1=A2=0 by hardware

//------R/W bit------------------
#define TWI_READ  1			   //reading from twi device 
#define TWI_WRITE 0            //writing to twi device 

#define SCL_CLOCK  100000UL	   //=Fscl, twi clock in Hz 100KHz freq kai kathorizei frequency epikoinwnias 

//Fscl=Fcpu/(16+2*TWBR0_VALUE*PRESCALER_VALUE), prescaler=1
#define TWBR0_VALUE ((F_CPU/SCL_CLOCK)-16)/2

// PCA9555 REGISTERS
typedef enum {   //se kathe addr antistixei enas reg
	REG_INPUT_0			 = 0,
	REG_INPUT_1			 = 1,
	REG_OUTPUT_0		 = 2,
	REG_OUTPUT_1	     = 3,
	REG_POLARITY_INV_0	 = 4,
	REG_POLARITY_INV_1	 = 5,
	REG_CONFIGURATION_0	 = 6,
	REG_CONFIGURATION_1	 = 7
} PCA9555_REGISTERS;

//AVR Master (transmitter and receiver)
//->transmitter otan grafei ston PCA9555 (Slave)
//->receiver otan diavazei apo PCA9555 (Slave)

//----------- Master Transmitter/Receiver -------------------
#define TW_START		0x08		//start
#define TW_REP_START	0x10		//repeated start

//---------------- Master Transmitter ----------------------
#define TW_MT_SLA_ACK	0x18		//ack apo siskevi otan stelnw addr
#define TW_MT_SLA_NACK	0x20		//nack apo siskevi otan stelnw addr
#define TW_MT_DATA_ACK	0x28		//ack otan steilw data


//---------------- Master Receiver ----------------
#define TW_MR_SLA_ACK	0x40		//ack apo siskevi otan stelnw addr
#define TW_MR_SLA_NACK	0x48		//nack apo siskevi otan stelnw addr
#define TW_MR_DATA_NACK	0x58		//an den dwsw nack termatizw me stop

#define TW_STATUS_MASK  0b11111000        //keep only TWS[7:3]
#define TW_STATUS (TWSR0 & TW_STATUS_MASK)

//-----------------initialize TWI clock--------------------
void twi_init(void)
{
	TWSR0 = 0;	// PRESCALER_VALUE=1
	TWBR0 = TWBR0_VALUE;
}

//diavazei teleutaio byte
//Read one byte from the twi device, read is followed by a stop condition
unsigned char twi_readNak(void)
{
	TWCR0 = (1<<TWINT) | (1<<TWEN);
	while(!(TWCR0 & (1<<TWINT)));
	return TWDR0;
}

// Issues a start condition and sends address and transfer direction.
// return 0 = device accessible, 1= failed to access device
//-----------------------stelnei START + ADDR(R/W) kai perimenei ACK--------------------------
unsigned char twi_start(unsigned char address)
{
	uint8_t   twi_status;
	
	// send START condition
	TWCR0 = (1<<TWINT) | (1<<TWSTA) | (1<<TWEN);
	
	// wait until transmission completed
	while(!(TWCR0 & (1<<TWINT)));
	
	// check value of TWI Status Register.
	twi_status = TW_STATUS & 0xF8;
	if ( (twi_status != TW_START) && (twi_status != TW_REP_START)) return 1;
	
	// send device address
	TWDR0 = address;
	TWCR0 = (1<<TWINT) | (1<<TWEN);
	
	// wail until transmission completed and ACK/NACK has been received
	while(!(TWCR0 & (1<<TWINT)));
	
	// check value of TWI Status Register.
	twi_status = TW_STATUS & 0xF8;
	if ( (twi_status != TW_MT_SLA_ACK) && (twi_status != TW_MR_SLA_ACK) ) //write or read mode 
	{
		return 1;
	}
	
	return 0;
}

//--------------------perimenei mexri na einai etoimi h siskevi gia na kanei start+addr----------------------
// Use ack polling to wait until device is ready
void twi_start_wait(unsigned char address)
{
	uint8_t   twi_status;
	
	while ( 1 )
	{
		// send START condition
		TWCR0 = (1<<TWINT) | (1<<TWSTA) | (1<<TWEN);
		
		// wait until transmission completed
		while(!(TWCR0 & (1<<TWINT)));
		
		// check value of TWI Status Register.
		twi_status = TW_STATUS & 0xF8;
		if ( (twi_status != TW_START) && (twi_status != TW_REP_START)) continue; //start while again
		
		// send device address
		TWDR0 = address;
		TWCR0 = (1<<TWINT) | (1<<TWEN);
		
		// wail until transmission completed
		while(!(TWCR0 & (1<<TWINT)));
		
		// check value of TWI Status Register.
		twi_status = TW_STATUS & 0xF8;
		if ( (twi_status == TW_MT_SLA_NACK )||(twi_status ==TW_MR_DATA_NACK) )
		{
			/* device busy, send stop condition to terminate write operation */
			TWCR0 = (1<<TWINT) | (1<<TWEN) | (1<<TWSTO);
			
			// wait until stop condition is executed and bus released
			while(TWCR0 & (1<<TWSTO));
			
			continue; //start while again
		}
		break; //sending addr was successful
	}
}

//------------------------stelnei 1 byte sti siskevi (write)----------------------------
//  Send one byte to twi device,  Return 0 if write successful or 1 if write failed
unsigned char twi_write( unsigned char data )
{
	// send data to the previously addressed device
	TWDR0 = data;
	TWCR0 = (1<<TWINT) | (1<<TWEN);
	
	// wait until transmission completed
	while(!(TWCR0 & (1<<TWINT)));
	if( (TW_STATUS & 0xF8) != TW_MT_DATA_ACK) return 1;
	return 0;
}

//-------------------repeated start + ADDR(R/W) kai perimenei ACK----------------------------
// Send repeated start condition, address, transfer direction
//Return:  0 device accessible 1 failed to access device
unsigned char twi_rep_start(unsigned char address)
{
	return twi_start( address );
}

//-----------------STOP----------------------
// Terminates the data transfer and releases the twi bus
void twi_stop(void)
{
// send stop condition
TWCR0 = (1<<TWINT) | (1<<TWEN) | (1<<TWSTO);

// wait until stop condition is executed and bus released
while(TWCR0 & (1<<TWSTO));
}

//-------------PCA9555_0 write kai read---------------------
void PCA9555_0_write(PCA9555_REGISTERS reg, uint8_t value)
{
	twi_start_wait(PCA9555_0_ADDRESS + TWI_WRITE);
	twi_write(reg);
	twi_write(value);
	twi_stop();
}
uint8_t PCA9555_0_read(PCA9555_REGISTERS reg)
{
	uint8_t ret_val;
	twi_start_wait(PCA9555_0_ADDRESS + TWI_WRITE); //prwta write+addr kai meta read
	twi_write(reg);
	twi_rep_start(PCA9555_0_ADDRESS + TWI_READ);
	ret_val = twi_readNak();
	twi_stop();
	return ret_val; 	
}

void write_2_nibbles(uint8_t x){
	int8_t low_bits;
	low_bits = PCA9555_0_read(REG_OUTPUT_0) & 0x0F;
	PCA9555_0_write(REG_OUTPUT_0 , low_bits | (x & 0xF0));  // PD0–PD3
	PCA9555_0_write(REG_OUTPUT_0 ,  (1 << PD3) | PCA9555_0_read(REG_OUTPUT_0));             // E = 1
	_NOP();
	_NOP();
	PCA9555_0_write(REG_OUTPUT_0 ,  ~(1 << PD3) & PCA9555_0_read(REG_OUTPUT_0));            // E = 0
	PCA9555_0_write(REG_OUTPUT_0 , low_bits | ((x<<4) & 0xF0));
	PCA9555_0_write(REG_OUTPUT_0 ,  (1 << PD3) | PCA9555_0_read(REG_OUTPUT_0));             // E = 1
	_NOP();
	_NOP();
	PCA9555_0_write(REG_OUTPUT_0 ,  ~(1 << PD3) & PCA9555_0_read(REG_OUTPUT_0));             // E = 0
}
void lcd_command(int x){
	PCA9555_0_write(REG_OUTPUT_0 ,  ~(1 << PD2) & PCA9555_0_read(REG_OUTPUT_0)); //set pd2 clear
	write_2_nibbles(x);
	_delay_us(250);
}
void lcd_clear_display(void){

	lcd_command(0x01);
	_delay_ms(5);
	
}

void lcd_data(uint16_t data)
{
	PCA9555_0_write(REG_OUTPUT_0 ,  (1 << PD2) | PCA9555_0_read(REG_OUTPUT_0));;     // RS = 1 -> Data
	write_2_nibbles(data);     //  2 nibbles
	_delay_us(250);            // wait 250 usec
	
}

/* Routine: usart_init 
Description: 
This routine initializes the 
usart as shown below. ------- INITIALIZATIONS ------- 
Baud rate: 9600 (Fck= 8MH) 
Asynchronous mode 
Transmitter on 
Reciever on 
Communication parameters: 8 Data ,1 Stop, no Parity -------------------------------- 
parameters: ubrr to control the BAUD. 
return value: None.*/ 
void usart_init(unsigned int ubrr){ 
UCSR0A=0; 
UCSR0B=(1<<RXEN0)|(1<<TXEN0); 
UBRR0H=(unsigned char)(ubrr>>8); 
UBRR0L=(unsigned char)ubrr; 
UCSR0C=(3 << UCSZ00); 
return; 
} 
/* Routine: usart_transmit 
Description: 
This routine sends a byte of data 
using usart. 
parameters: 
data: the byte to be transmitted 
return value: None. */ 
void usart_transmit(uint8_t data){ 
while(!(UCSR0A&(1<<UDRE0))); 
UDR0=data; 
} 
/* Routine: usart_receive 
Description: 
This routine receives a byte of data 
from usart. 
parameters: None. 
return value: the received byte */ 
uint8_t usart_receive(){ 
while(!(UCSR0A&(1<<RXC0))); 
return UDR0; 
}

void lcd_print_string(char *str){
    while(*str != 0){
        lcd_data(*str);
        str++;
    }
}

void usart_transmit_string(char *str){
    while(*str!=0){
        usart_transmit(*str);
        str++;
    }
}

void usart_receive_string(char *buffer, int max_len){
    int i = 0;
    
    while(i < max_len - 1){ 
        buffer[i] = usart_receive();
         if(buffer[i] == '\n'){ 
            break; 
        }   
        i++;
    }
    buffer[i] = '\0';
}

uint8_t one_wire_reset(){

    DDRD|=1<<PD4; //set PD4 as output
    PORTD&=~(1<<PD4); //PD4=0
    _delay_us(480); //480us delay reset pulse

    DDRD&=~(1<<PD4); //set PD4 as input
    PORTD&=~(1<<PD4); //disable pull-ups
    _delay_us(100); //wait 100us for connected devices to transmit the presence pulse

    uint8_t input=PIND; //read input
    _delay_us(380); //380us delay 

    uint8_t connected_device=0;
    if(!(input&(1<<PD4))) connected_device=1;

    return connected_device;

}

uint8_t one_wire_receive_bit (){
    
    DDRD|=1<<PD4; //set PD4 as output
    PORTD&=~(1<<PD4); //PD4=0
    _delay_us(2); //delay 2us

    DDRD&=~(1<<PD4); //set PD4 as input
    PORTD&=~(1<<PD4); //disable pull ups
    _delay_us(10); //wait 10us
    
    uint8_t received_bit=0;
    if(PIND&(1<<PD4)) received_bit=1;

    _delay_us(49); //delay 49us

    return received_bit;    
        
}

void one_wire_transmit_bit(uint8_t bit){
    
    DDRD|=1<<PD4; //set PD4 as output
    PORTD&=~(1<<PD4); //PD4=0
    _delay_us(2); //delay 2us

    if(bit) 
        PORTD|=1<<PD4; //PD4=1
    else
         PORTD&=~(1<<PD4); //PD4=0

    _delay_us(58); //wait 58us for connected device to sample the line

    DDRD&=~(1<<PD4); //set PD4 as input
    PORTD&=~(1<<PD4); //disable pull ups

    _delay_us(1); //recovery time
    
}
uint8_t one_wire_receive_byte(){
    uint8_t received_byte=0;
    uint8_t received_bit;
    for(int i=0;i<8;i++){
        received_bit=one_wire_receive_bit(); //receive each bit
        received_byte=received_byte >> 1; //lsr in order to make the mask
        if(received_bit){
            received_byte|=0b10000000; //or
        }
    } 
        return received_byte;
}     
void one_wire_transmit_byte(uint8_t byte_to_send){
    uint8_t transmit_bit;
    for(int i=0;i<8;i++){
        transmit_bit=byte_to_send & 0x01; //apomonwsi lsb
        one_wire_transmit_bit(transmit_bit); //stelnoume to bit
        byte_to_send=byte_to_send >> 1;  //shift gia na erthei h seira tou epomenou

    }
}

int16_t temperature_measure(){  

    int16_t temperature=0;
    if(!(one_wire_reset())){
        return 0x8000;  //error handler
    }
    one_wire_transmit_byte(0xCC);
    one_wire_transmit_byte(0x44); //xekinaei metrisi
    while(!one_wire_receive_bit()); //gia oloklirosi metrisis 
    if(!(one_wire_reset())){ //nea arxikopoiisi
        return 0x8000;
    } 
    one_wire_transmit_byte(0xCC);
    one_wire_transmit_byte(0xBE);
    temperature=one_wire_receive_byte();
    temperature|=(int16_t)one_wire_receive_byte()<<8; //enwnw lsb kai msb
    return temperature;

}

uint8_t scan_row(int numline){
	uint8_t check_input=0;
	if(numline==0){
		PCA9555_0_write(REG_OUTPUT_1,0b11111110); //output mono h  I01_0
		_delay_ms(1);
		check_input=PCA9555_0_read(REG_INPUT_1) & 0xF0;
	}
	else if(numline==1){
		PCA9555_0_write(REG_OUTPUT_1,0b11111101);//output mono h  I01_1
		_delay_ms(1);
		check_input=PCA9555_0_read(REG_INPUT_1) & 0xF0;
	}
	else if(numline==2){
		PCA9555_0_write(REG_OUTPUT_1,0b11111011);//output mono h  I01_2
		_delay_ms(1);
		check_input=PCA9555_0_read(REG_INPUT_1) & 0xF0;
	}
	else if(numline==3){
		PCA9555_0_write(REG_OUTPUT_1,0b11110111);//output mono h  I01_3
		_delay_ms(1);
		check_input=PCA9555_0_read(REG_INPUT_1) & 0xF0;
	}
	return check_input>>4;
}
uint16_t scan_keypad(){
	uint16_t full_mask=0x0000;
	for(int i=0;i<4;i++){
		full_mask|=(scan_row(i)<<(4*i));
	}
	return full_mask;
}
uint16_t scan_keypad_pressed(){
	uint16_t pressed_keys_tempo=~scan_keypad(); //1 patima diladi high active
	_delay_ms(20);
	pressed_keys_tempo&=~scan_keypad(); //tropopoisi 1 patima
	return pressed_keys_tempo;
}
int keypad_to_ascii(void){
	int patimeno = scan_keypad_pressed();
	int ascicode = 0;                         // default 0

	// 4 LSBs  ->  * 0 # D
	if      (patimeno == 0x0001) ascicode = '*';
	else if (patimeno == 0x0002) ascicode = '0';
	else if (patimeno == 0x0004) ascicode = '#';
	else if (patimeno == 0x0008) ascicode = 'D';

	// next 4 bits -> 7 8 9 C
	else if (patimeno == 0x0010) ascicode = '7';
	else if (patimeno == 0x0020) ascicode = '8';
	else if (patimeno == 0x0040) ascicode = '9';
	else if (patimeno == 0x0080) ascicode = 'C';

	// next 4 bits -> 4 5 6 B
	else if (patimeno == 0x0100) ascicode = '4';
	else if (patimeno == 0x0200) ascicode = '5';
	else if (patimeno == 0x0400) ascicode = '6';
	else if (patimeno == 0x0800) ascicode = 'B';

	// MSBs-> 1 2 3 A
	else if (patimeno == 0x1000) ascicode = '1';
	else if (patimeno == 0x2000) ascicode = '2';
	else if (patimeno == 0x4000) ascicode = '3';
	else if (patimeno == 0x8000) ascicode = 'A';
	return ascicode;   // 0 if nothing pressed
}
